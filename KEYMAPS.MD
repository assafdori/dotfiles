# Tmux + Nvim keymaps that I learn as I go.

NOTE: These are custom keymaps for Nvim and Tmux.
-------------------------------------------------
Prefix = ^a
n = space(?)


#### buffer management
n + tl = next buffer
n + th = previous buffer
n + td = delete buffer


#### tmux management
^a + ^C = new tmux window
^a + 1-9 = select tmux window
^a + s = split tmux pane
^a + v = vsplit tmux pane
^a + c = close tmux pane
^a + S = open tmux sessions
^a + r = rename window
^a + d = detach tmux
^a + $ = rename session


#### nvim commands
space + ng = neogit
space + B = line beginning
space + E = line end
space + WW = write file
space + st = telescope :notes
space + tw = twilight mode

shift + ( = quick up
shift + ) = quick down
